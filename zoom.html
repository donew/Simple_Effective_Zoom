$(".zoom-container").mousemove(function (e) {
        var $img_wid = $(this).find("img");
        var imgCtnWid = $(this).width();
        var imgCtnHet = $(this).height();
        var he = $img_wid.height() - imgCtnHet;
        var wi = $img_wid.width() - imgCtnWid;
        var parentOffset = $(this).offset();
        var relX = e.pageX - parentOffset.left;
        var relY = e.pageY - parentOffset.top;       
        if ($img_wid.width() <= imgCtnWid) { }
        else {
            var wi_all = relX / imgCtnWid * 100;
            var perXimg = wi * wi_all / 100;
        }
        if ($img_wid.height() <= imgCtnHet) {  }
        else {

            var top_all = relY / imgCtnHet * 100;
            var perimg = he * top_all / 100;
        }
        $img_wid.css({ "left": -perXimg, "top": -perimg });
    });
});
